int main(int argc, char** argv){
    if(argc != 2) {
        usage();
        return 1;
    }


    VideoCapture cap(argv[1]);
    if(!cap.isOpened()) {
        cap.release();
        cerr << "Unable to open video file"endl;
        return 1;
    }

    return 0;
}
